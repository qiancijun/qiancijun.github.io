int maxScore(string s) {
    int result = 0;
    for (int p = 1; p <= s.size() - 1; p++){
        int left = 0;
        int right = 0;
        for (int i = 0; i < p; i++){
            if (s[i] == '0')
                left++;
        }
        for (int j = p; j < s.size(); j++){
            if (s[j] == '1')
                right++;
        }
        result = max(result, left + right);
    }
    return result;
}

int maxScore(vector<int>& cardPoints, int k) {
    int len = cardPoints.size();
    int result = 0;
    int sum = 0;
    int i, j = len - 1;
    for (i = 0; i < k; i++){
        sum += cardPoints[i];
    }
    i--;
    result = sum;
    while (i >= 0 && j >= 0){
        sum = sum - cardPoints[i--] + cardPoints[j--];
        result = max(result, sum);
    }
    return result;
}
