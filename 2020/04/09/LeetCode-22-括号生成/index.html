思路

由于这里要求出所有解的情况，而一般使用DFS就能求所有解，思路也很简单，先放左括号再放右括号。

代码

vector<string> generateParenthesis(int n) {
    vector<string> result;
    if(n == 0){
        return result;
    }
    dfs(result, "", n, n, n);
    return result;
}
void dfs(vector<string> &result, string list, int left, int right, int n){
    if(left == 0 && right == 0){
        result.push_back(list);
        return;
    }
    if(left > 0){
        dfs(result, list + '(', left - 1, right, n);
    }
    if(right > left){
        dfs(result, list + ')', left, right - 1, n);
    }
    return;
}
