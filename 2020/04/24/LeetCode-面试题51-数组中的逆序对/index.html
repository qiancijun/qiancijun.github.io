思路

由于题目给了限制，第一时间想到的双重循环暴力解法一定会超时。这道题典型运用了分治思想，在排序过程中把能把数组化成子数组且能判断大小的只有归并排序，所以我们在对数组排序的过程中直接去比较大小来计算逆序对的个数。

代码

int reversePairs(vector<int>& nums) {
    int len = nums.size();
    if (len < 2)
        return 0;
    vector<int> temp(len, 0);
    return MergeSort(nums, 0, len - 1, temp);   
}
int MergeSort(vector<int> &nums, int Left, int Right, vector<int> &temp){
    if (Left == Right)
        return 0;
    int Mid = Left + (Right - Left) / 2;
    int LeftMerge = MergeSort(nums, Left, Mid, temp);
    int RightMerge = MergeSort(nums, Mid + 1, Right, temp);
    int LastMerge = MergeAndCount(nums, Left, Mid, Right, temp);
    return LeftMerge + RightMerge + LastMerge;
}
int MergeAndCount(vector<int> &nums, int Left, int Mid, int Right, vector<int> &temp){
    for (int i = Left; i <= Right; i++){
        temp[i] = nums[i];
    }
    int i = Left;
    int j = Mid + 1;
    int count = 0;
    for (int k = Left; k <= Right; k++){
        if (i == Mid + 1){
            nums[k] = temp[j++];
        } else if (j == Right + 1){
            nums[k] = temp[i++];
        } else if (temp[i] <= temp[j]){
            nums[k] = temp[i++];
        } else {
            nums[k] = temp[j++];
            count += (Mid - i + 1);
        }
    }
    return count;
}
