5380. 数组中的字符串匹配

思路

双重循环硬解，利用好语言特性

标签

字符串

代码

vector<string> stringMatching(vector<string>& words) {
    vector<string> result;
    for (int i = 0; i < words.size(); i++){
        for (int j = 0; j < words.size(); j++){
            if (i == j || words[i].size() > words[j].size())
                continue;
            if (words[j].find(words[i]) != std::string::npos){
                result.push_back(words[i]);
                break;
            }
        }
    }
    return result;
}

5381. 查询带键的排列

思路

暴力法。

标签

数组

代码

vector<int> processQueries(vector<int>& queries, int m) {
    vector<int> result;
    vector<int> p(m, 0);
    for (int i = 0; i < m; i++){
        p[i] = i + 1;
    }
    for (int i = 0; i < queries.size(); i++){
        for (int j = 0; j < m; j++){
            if (queries[i] == p[j]){
                for (int k = j; k > 0; k--)
                    swap(p[k], p[k - 1]);
                result.push_back(j);
                break;
            }
        }
    }
    return result;
}

1410. HTML 实体解析器

思路

用一个map就能解决

标签

栈 字符串

代码

string entityParser(string text) {
    map<string, string> m = {
        {"&quot;", "\""},
        {"&apos;", "'"},
        {"&amp;", "&"},
        {"&gt;", ">"},
        {"&lt;", "<"},
        {"&frasl;", "/"}
    };

    int len = text.size();
    for (int i = 0; i < len; i++){
        if (text[i] == '&'){
            int j; 
            string key = "";
            for (j = i; j < len; j++){
                if (text[j] == ';'){
                    key += ';';
                    break;
                } else {
                    key += text[j];
                }
            }
            if (m.count(key)){
                text.replace(i, key.size(), m[key]);
            }
        }
        len = text.size();
    }
    return text;
}

5383. 给 N x 3 网格图涂色的方案数

思路

去考虑ABA和ABC色块的关系

标签

动态规划

代码

const int mod = 1000000007;
int numOfWays(int n) {
    int result = 0;
    vector<vector<long>> dp(n, vector<long>(2, 0));
    dp[0][0] = 6;
    dp[0][1] = 6;//状态初始化
    /*
    动态转移方程dp[ABA] = 3 * dp[i-1][j] + 2 * dp[i - 1][j + 1]
    dp[ABC] = 2 * dp[i-1][j] + 2 * dp[i - 1][j - 1]
    */
    for (int i = 1; i < n; i++){
        for (int j = 0; j < 2; j++){
            if (j == 0){
                dp[i][0] = (3 * dp[i - 1][0] + 2 * dp[i - 1][1]) % mod;
            }
            else{
                dp[i][1] = (2 * dp[i - 1][0] + 2 * dp[i - 1][1]) % mod;
            }
        }
    }
    result = (dp[n-1][0] + dp[n-1][1]) % mod;
    return result;
}
