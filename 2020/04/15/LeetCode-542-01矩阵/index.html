思路

先遍历一遍矩阵，把0全部找出来，然后BFS模板一套，完事

代码

vector<vector<int>> updateMatrix(vector<vector<int>>& matrix) {
    int m = matrix.size(); //row
    int n = matrix[0].size(); //col
    vector<int> di = {-1, 0, 1, 0};
    vector<int> dj = {0, -1, 0, 1}; //方向数组
    vector<vector<int>> result(m, vector<int>(n, -1));
    queue<vector<int>> q;
    for (int i = 0; i < m; i++){
        for (int j = 0; j < n; j++){
            if (matrix[i][j] == 0){
                result[i][j] = 0;
                q.push({i, j});
            }
        }
    }
    while (!q.empty()){
        int time = q.size();
        while (time--){
            vector<int> curr = q.front();
            q.pop();
            int x = curr[0];
            int y = curr[1];
            for (int d = 0; d <= 3; d++){
                int px = x + di[d];
                int py = y + dj[d];
                if (px >= 0 && px < m && py >= 0 && py < n && result[px][py] == -1){
                    result[px][py] = result[x][y] + 1;
                    q.push({px, py});
                }
            }
        }
    }
    return result;
}
