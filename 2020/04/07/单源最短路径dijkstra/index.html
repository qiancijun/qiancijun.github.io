前言

在做无权值图的单源最短路径问题中，我们能使用DFS或者BFS算法就能解决问题。而在有权值的图中，就需要一点特殊手段处理了，本文主要介绍dijkstra算法。在这里我们所讨论的图都是有向图，没有负值圈。

思路

Dijkstra算法是典型的 $\color{red}{单源最短路径算法}$， 用于计算一个节点到其他所有节点的最短路径。主要特点是 $\color{red}{以起始点为中心向外层扩展，直到终点为止}$

算法实现

假设所给图是带权有向图

假设一个集合S，S中只有一个源点，每次求得一条最短路径后，就把该顶点加入到S集合中，直到所有顶点都加入该集合。思想有点类似于贪心算法。对于每次取最小路径这个问题，我们可以使用一个特殊的数据结构，优先队列(priority_queue)。优先队列会把权值小的插队到权值大的前面，这样一来，我们能够保证每一步走的都是最短的，但同时我们需要记录该节点是否被走过，以免被覆盖。

图的描述

这里我选择的是用散列表来表示一个图,key是源节点,value是指向的节点，形式为 源节点： [指向的节点， 权值]

广度优先搜索

这是一个标准的BFS模板，一个队列，节点不断入队，稍作修改。CPP的优先队列默认是大顶堆，这里还需要重写一下比较方式，直接利用了一个lambda(c++11)或者传入一个compare函数。用一个visited数组标记是否被访问过，-1代表没被访问，如果访问后，把值改为最短路径和。用一个path数组记录该节点是怎么走过来的，下标对于节点编号，值对应了从哪个节点过来，直到一直追溯到0，说明就已经找到了源节点。再用一个堆栈逆序打印出顺序。

#include <iostream>
#include <vector>
#include <unordered_map>
#include <functional>
#include <queue>
#include <stack>

using namespace std;

void dijkstra(vector<vector<int>> nums, int K, vector<int> &vis, vector<int> &path){
    unordered_map<int, vector<vector<int>>> graph; //图
    for(auto point: nums){
        graph[point[0]].push_back({point[1], point[2]}); //源点，{节点， 权值}
    }

    //优先队列，重载>运算符
    priority_queue<vector<int>, vector<vector<int>>, function<bool(vector<int> &v1, vector<int> &v2)>> pq([](vector<int> &v1, vector<int> &v2)->bool{
        return v1[1] > v2[1];
    });

    pq.push({K, 0});

    while (!pq.empty()){
        vector<int> cur = pq.top();
        pq.pop();

        //节点已经被访问过
        if (vis[cur[0]] != -1)
            continue;
        vis[cur[0]] = cur[1];
        for (auto next: graph[cur[0]]){
            if (vis[next[0]] != -1){
                continue;
            }
            path[next[0]] = cur[0];
            pq.push({next[0], cur[1] + next[1]});
        } //刷新一下图
    }
    return;
}
 //打印出路径，即 是什么走到该节点的
void find_origin(vector<int> path, int K){
    stack<int> s;
    int i = K;
    s.push(K);
    while (path[i] != 0)
    {
        s.push(path[i]);
        i = path[i];
    }
    cout << K << "'s origin:";
    while (!s.empty())
    {
        cout << s.top() << " ";
        s.pop();
    }
    cout << endl;
}

int main(void){
    const int SIZE = 6;
    vector<vector<int>> nums = {
        {1, 2, 1},
        {1, 4, 2},
        {2, 3, 5},
        {2, 4, 1},
        {4, 3, 2},
        {3, 6, 1},
        {4, 5, 1},
        {5, 6, 4}
    };
    vector<int> vis(SIZE + 1, -1);
    vector<int> path(SIZE + 1, 0);
    dijkstra(nums, 1, vis, path);
    for (int i = 1; i < vis.size(); i++){
        cout << vis[i] << " ";
    }
    cout << endl;
    for(int i = 1; i < path.size();i++){
        find_origin(path, i);
    }
    getchar();
    return 0;
}
